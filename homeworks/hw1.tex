\begin{exercise}\label{exer:image-zero-section-is-open}
Let $\pi \colon S \to X$ be any sheaf and $z \colon X \to S$ the zero section. Then $\im(z) \subseteq S$ is open.
\end{exercise}


\begin{exercise} Show that every morphism between sheaves of abelian groups is a local homeomorphism.
\end{exercise}

\begin{exercise} Let $f \colon S \to S'$ be a morphism of sheaves over $X$. Verify if $S_x \to S_x'$ is an isomorphism of abelian groups for each $x\in X$ then $f$ is a homeomorphism (i.e. an isomorphism of sheaves).
\end{exercise}

\begin{exercise} Let $f \colon F \to G$ be a morphism of presheaves on $X$ satisfying locality and gluing. Verify if the induced map on stalks $f_x \colon F_x \to G_x$ is an isomorphism for each $x\in X$ then $f$ is a natural isomorphism.
\end{exercise}
